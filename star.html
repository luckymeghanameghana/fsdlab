styles.css .star-rating { display: flex; font-size: 2rem; cursor: pointer; } .star { color: #ddd; transition: color 0.2s ease; } .star.hover, .star.selected { color: gold; } #rating-value { margin-top: 10px; font-size: 1.2rem; } script.js $(document).ready(function() { var rating = 0; // Mouse enter event to highlight stars $('.star').on('mouseenter', function() { var value = $(this).data('value'); $('.star').each(function() { if ($(this).data('value') <= value) { $(this).addClass('hover'); } else { $(this).removeClass('hover'); } }); }); // Mouse leave event to remove highlights $('.star').on('mouseleave', function() { $('.star').removeClass('hover'); }); // Click event to set rating $('.star').on('click', function() { rating = $(this).data('value'); $('#rating-value').text('Rating: ' + rating); $('.star').each(function() { if ($(this).data('value') <= rating) { $(this).addClass('selected'); } else { $(this).removeClass('selected'); } }); });
